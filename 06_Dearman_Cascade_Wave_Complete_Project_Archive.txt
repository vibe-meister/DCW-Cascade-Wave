# Core Simplified Simulator (axisymmetric)
import numpy as np

class DIMFluidSimulator:
    def __init__(self, Re=2000, nr=50, nz=100, H=2.5, dim_on=True):
        self.Re = Re
        self.dr = 1 / (nr - 1)
        self.dz = H / (nz - 1)
        self.r = np.linspace(0, 1, nr)
        self.z = np.linspace(0, H, nz)
        self.psi = np.zeros((nr, nz))
        self.gamma = np.zeros((nr, nz))
        self.dim_on = dim_on
        self.res_freqs = [3,6,9,12,18,24,48,75]
        self.larmor_omega = 0.1

    def resonance_term(self, field):
        if not self.dim_on: return 0
        # Simplified mock amplification
        return 0.05 * np.sin(2*np.pi*np.array(self.res_freqs)[:,None,None] * field)

    def step(self, dt=0.01):
        # Mock advection/diffusion + DIM terms
        adv = np.random.rand(*self.gamma.shape) * 0.01  # placeholder
        diff = (1/self.Re) * (np.roll(self.gamma,1,axis=0) - 2*self.gamma + np.roll(self.gamma,-1,axis=0))/self.dr**2
        res = self.resonance_term(self.gamma)
        self.gamma += dt * (-adv + diff + res)

    def run(self, t_max=50):
        for _ in range(int(t_max/0.01)):
            self.step()
        return np.max(np.abs(self.gamma)), np.var(self.gamma.flatten()[-1000:])

# 3D Amplitude Equation
from scipy.integrate import odeint

def hopf_dim(y, t, mu, omega, l, dim_on):
    A = y[0] + 1j*y[1]
    dA = (mu + 1j*omega)*A - l*abs(A)**2*A
    if dim_on:
        for k in [1,2,3]:
            dA += 0.05 * np.exp(1j*2*np.pi*0.3*k*t)
    dAdt = [dA.real, dA.imag]
    return dAdt

# Example usage
t = np.linspace(0,2000,20000)
y0 = [0.1, 0.0]
